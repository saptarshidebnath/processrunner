42  /**
43   * The {@link Configuration} object holds the configuration.
44   *
45   * <p>The created configuration is used by the {@link Runner}.
46   */
47  public class Configuration {
48  
49    private static final Logger logger = LoggerFactory.getLogger(Configuration.class);
50    private final String interpreter;
51    private final String command;
52    private final Path workingDir;
53    private final File masterLogFile;
54    private final Charset charset;
55    private final boolean autoDeleteFileOnExit;
56    private final boolean enableLogStreaming;
57  
58    /**
59     * Constructor to set the configuration to be consumed by {@link Runner}.
60     *
61     * @param interpreter : sets the {@link String} command interpreter like /bin/bash in unix
62     * @param command : set the actual {@link String} command to be executed
63     * @param workingDir : sets the working directory in {@link File} format
64     * @param masterLogFile : {@link File} where the log data will be stored.
65     * @param charset : a reference of which {@link Charset} to use while writing the {@link
66     *     Configuration#masterLogFile}
67     * @param autoDeleteFileOnExit : set the flag to denote if the sysout and the syserror {@link
68     *     File} going to be auto deleted on exit.
69     * @param enableLogStreaming : enable lor disable log streaming by passing a @{@link Boolean}
70     *     value
71     */
72    Configuration(
73        final String interpreter,
74        final String command,
75        final Path workingDir,
76        final File masterLogFile,
77        final Charset charset,
78        final boolean autoDeleteFileOnExit,
79        final boolean enableLogStreaming) {
80      this.interpreter = interpreter.trim();
81      this.command = command.trim();
82      this.workingDir = workingDir;
83      this.autoDeleteFileOnExit = autoDeleteFileOnExit;
84      this.masterLogFile = masterLogFile;
85      if (this.autoDeleteFileOnExit) {
86        this.masterLogFile.deleteOnExit();
87      }
88      this.enableLogStreaming = enableLogStreaming;
89      this.charset = charset;
90      logger.debug("Process Runner Configuration : {}", this);
91    }

189   /**
190    * {@link ConfigBuilder} helps in composing a {@link Configuration}.
191    *
192    * <p>The {@link Configuration} can be composed using the {@link ConfigBuilder}.
193    */
194   public static class ConfigBuilder {
195 
196     private static final Logger logger = LoggerFactory.getLogger(ConfigBuilder.class);
197     private final String interpreter;
198     private final String command;
199     private final List<String> comamndParams;
200     private Path workingDir;
201     private File masterLogFile;
202     private boolean autoDeleteFileOnExit;
203     private boolean logStreamingEnabled;
204     private Charset charset;
205 
206     /**
207      * The constructor for {@link ConfigBuilder}. The only required parameters are the {@link
208      * String} interpreter and the {@link String} command.
209      *
210      * @param interpreter as {@link String} for example "bash", "cmd.exe /c"
211      * @param command as {@link String} for example "echo 'I solemnly swear I am up to no good'"
212      * @throws ProcessConfigurationException is thrown if the interpreter or the command is null or
213      *     empty {@link String}
214      */
215     public ConfigBuilder(String interpreter, String command) throws ProcessConfigurationException {
216       if (interpreter == null || interpreter.trim().length() == 0) {
217         throw new ProcessConfigurationException(
218             "Command RunnerImpl Interpreter is set '"
219                 + interpreter
220                 + "'. Need a valid command runner interpreter as /bin/bash in unix");
221       } else if (null == command || 0 == command.trim().length()) {
222         throw new ProcessConfigurationException(
223             "Command is set '" + command + "'. Need a valid command like 'echo Hello World'");
224       }
225       logger.trace("Interpreter and command validation passed");
226       this.interpreter = interpreter;
227       this.command = command;
228       this.comamndParams = new ArrayList<>();
229       this.logStreamingEnabled = false;
230     }
231 
232     /**
233      * Add a {@link String} parameter the the command to be executed.
234      *
235      * @param param as {@link String}
236      * @return the {@link ConfigBuilder}
237      * @throws ProcessConfigurationException if the parameter is null or an empty String.
238      */
239     public ConfigBuilder setParam(String param) throws ProcessConfigurationException {
240       if (param == null || param.length() == 0) {
241         throw new ProcessConfigurationException("Param is either null or empty.");
242       }
243       logger.trace("Param parameter passed validation");
244       comamndParams.add(param);
245       return this;
246     }
247 
248     /**
249      * Add a {@link List} parameter of type {@link String} to the command to be executed.
250      *
251      * @param paramList as {@link List} of type {@link String}. The order of the parameters are
252      *     maintained.
253      * @return the {@link ConfigBuilder}
254      * @throws ProcessConfigurationException if the parameter is null or an empty String.
255      */
256     public ConfigBuilder setParamList(List<String> paramList) throws ProcessConfigurationException {
257       if (paramList == null || paramList.isEmpty()) {
258         throw new ProcessConfigurationException("Param list is either null or empty.");
259       }
260       logger.trace("Param list parameter passed validation");
261       comamndParams.addAll(paramList);
262       return this;
263     }
264 
265     /**
266      * Set the current working Directory as {@link Path}.
267      *
268      * @param workingDir accepts {@link Path} current working directory
269      * @return the {@link ConfigBuilder}
270      * @throws ProcessConfigurationException if the working directory is not a directory or it
271      *     doesn't exist.
272      */
273     public ConfigBuilder setWorkigDir(Path workingDir) throws ProcessConfigurationException {
274       if (!workingDir.toFile().exists() || !workingDir.toFile().isDirectory()) {
275         throw new ProcessConfigurationException(
276             "Command's current directory is set '"
277                 + workingDir.toAbsolutePath().toString()
278                 + "'. Either the Directory doesn't exist or is not a directory at all");
279       }
280       logger.trace("Working dir parameter passed validation");
281       this.workingDir = workingDir;
282       return this;
283     }
284 
285     /**
286      * Set the master log file as {@link File}.
287      *
288      * <p>Master log file is a {@link File} where all the logs are stored by default. If the {@link
289      * Configuration#masterLogFile} is not set, the logs are not store anywhere and are discarded.
290      *
291      * @param masterLogFile accepts a {@link File} reference as master log file.
292      * @param autoDeleteFileOnExit accepts a {@link Boolean} flag to determine if the master log
293      *     file is going to be auto deleted or not on JVM exit.
294      * @return the {@link ConfigBuilder}
295      * @throws ProcessConfigurationException is thrown if the {@link File} object passed refers to a
296      *     * directory or is a read only file
297      */
298     public ConfigBuilder setMasterLogFile(File masterLogFile, boolean autoDeleteFileOnExit)
299         throws ProcessConfigurationException {
300       return this.setMasterLogFile(
301           masterLogFile, autoDeleteFileOnExit, ProcessRunnerConstants.UTF_8);
302     }
303 
304     /**
305      * Set the master log file as {@link File} with {@link Boolean} autodelete and {@link Charset}.
306      *
307      * <p>Master log file is a {@link File} where all the logs are stored by default. If the {@link
308      * Configuration#masterLogFile} is not set, the logs are not store anywhere and are discarded.
309      *
310      * @param masterLogFile accepts a {@link File} reference as master log file.
311      * @param autoDeleteFileOnExit accepts a {@link Boolean} flag to determine if the master log
312      *     file is going to be auto deleted or not on JVM exit.
313      * @param charset a reference of {@link Charset} class to set in which chaset the output file is
314      *     going to be written.
315      * @return the {@link ConfigBuilder}
316      * @throws ProcessConfigurationException is thrown if the {@link File} object passed refers to a
317      *     directory or is a read only file
318      */
319     public ConfigBuilder setMasterLogFile(
320         File masterLogFile, boolean autoDeleteFileOnExit, Charset charset)
321         throws ProcessConfigurationException {
322       if (masterLogFile.isDirectory()) {
323         throw new ProcessConfigurationException(
324             "Master log file : "
325                 + masterLogFile.getAbsolutePath()
326                 + " is a directory. It must be a writable file");
327       } else if (!masterLogFile.canWrite()) {
328         throw new ProcessConfigurationException(
329             "Master log file : "
330                 + masterLogFile.getAbsolutePath()
331                 + " is a readonly. It must be a writable file");
332       }
333 
334       this.masterLogFile = masterLogFile;
335       this.autoDeleteFileOnExit = autoDeleteFileOnExit;
336       this.charset = charset;
337       return this;
338     }
339 
340     /**
341      * Enable or disable log streaming. Please note that streaming will happen or not depends if the
342      * SLF4j finding the logging framework binding and the current log level set for that particular
343      * logging framework. All the log streaming will happen in the level equivalent to {@link
344      * java.util.logging.Level#INFO}
345      *
346      * @param logStreamingEnabled a reference of {@link PrintStream}
347      * @return the {@link ConfigBuilder}
348      */
349     public ConfigBuilder enableLogStreaming(boolean logStreamingEnabled) {
350       logger.trace("Setting log streaming as per request");
351       this.logStreamingEnabled = logStreamingEnabled;
352       return this;
353     }
354 
355     /**
356      * Builds the {@link Configuration} object and returns it back.
357      *
358      * @return a reference to the {@link Configuration} object created.
359      */
360     public Configuration build() {
361       logger.trace("Building configuration");
362       String commandWithParam =
363           new StringJoiner(ProcessRunnerConstants.SPACE_STR)
364               .add(command)
365               .add(
366                   comamndParams
367                       .stream()
368                       .collect(Collectors.joining(ProcessRunnerConstants.SPACE_STR)))
369               .toString();
370       logger.trace("Command to be executed : {}", commandWithParam);
371       return new Configuration(
372           interpreter,
373           commandWithParam,
374           workingDir,
375           masterLogFile,
376           charset,
377           autoDeleteFileOnExit,
378           logStreamingEnabled);
379     }
380   }
