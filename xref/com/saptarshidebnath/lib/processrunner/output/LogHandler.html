30  /**
31   * Writes the process logs as single line json of format {@link OutputRecord} to the file as
32   * configured in {@link Configuration#getMasterLogFile()}. If {@link
33   * Configuration#getMasterLogFile()} returns null, the logs will not be written.
34   */
35  public class LogHandler {
36  
37    private Logger logger = LoggerFactory.getLogger(LogHandler.class);
38    private PrintWriter printWriter;
39    private BlockingQueue<OutputRecord> queue;
40    private boolean streamingEnabled;
41    private ArrayList<Future> inputStreamReadingThreads;
42    private ExecutorService executorService;
43    private Future diskWritingThread;
44    private String processConfigrationAsString;
45    private File masterLogFile;
46    private boolean logsNeedTobeWritten;
47    private Process process;
48    private Configuration configuration;

50    /**
51     * The construcor of the class {@link LogHandler}.
52     *
53     * <p>The class receives a reference of the {@link Process} and {@link Configuration}. The class
54     * then internally takes care of creating all the process output reading threads i.e. both {@link
55     * OutputSourceType#SYSERR} and {@link OutputSourceType#SYSOUT} and writing the same to the disk
56     * in separate thread. The class also provides a blocking method {@link
57     * LogHandler#waitForShutdown()} so that you can wait for all logs to be streamed, written to disk
58     * or both.
59     *
60     * <p>The class internally uses an executor service from {@link
61     * Threadify#getProcessRunnerExecutorService()}
62     *
63     * @param process a object of type {@link Process}
64     * @param configuration a reference of type {@link Configuration}
65     */
66    public LogHandler(Process process, Configuration configuration) {
67      this.process = process;
68      this.configuration = configuration;
69      this.masterLogFile = this.configuration.getMasterLogFile();
70      this.streamingEnabled = this.configuration.isEnableLogStreaming();
71      this.logsNeedTobeWritten = this.configuration.getMasterLogFile() != null;
72      this.processConfigrationAsString = this.configuration.toString();
73    }
74  
75    public LogHandler start() throws FileNotFoundException {
76  
77      boolean logsNeedTobeRead = logsNeedTobeWritten || streamingEnabled;
78      if (logsNeedTobeRead) {
79        this.queue = new LinkedBlockingQueue<>(ProcessRunnerConstants.CACHE_SIZE);
80        executorService = new Threadify().getProcessRunnerExecutorService();
81        if (streamingEnabled) {
82          logger.info("Logs will be streamed on real time.");
83        } else {
84          logger.warn("Logs streaming disabled.");
85        }
86        this.inputStreamReadingThreads = new ArrayList<>();
87        //
88        // Track SYSOUT
89        //
90        this.saveInpuStreamToDisk(process.getInputStream(), OutputSourceType.SYSOUT);
91        //
92        // Track SYSERR
93        //
94        this.saveInpuStreamToDisk(process.getErrorStream(), OutputSourceType.SYSERR);
95        //
96        // Write content to DISK
97        //
98        if (logsNeedTobeWritten) {
99          this.printWriter =
100             new PrintWriter(
101                 new OutputStreamWriter(
102                     new FileOutputStream(masterLogFile), this.configuration.getCharset()));
103         this.diskWritingThread = executorService.submit(this::writeToDisk);
104       } else {
105         logger.warn(
106             "Logs not written to file as per configuration : {}", processConfigrationAsString);
107       }
108       //
109       // Mark for shutdown after execution is complete.
110       //
111       this.executorService.shutdown();
112       logger.debug("Created LogHandler. Tracking SYSOUT and SYSERROR");
113     } else {
114       logger.warn("Log Streaming is not enabled and Master logfile not set. Discarding logs.");
115       logger.warn("Configuration received : {}", processConfigrationAsString);
116     }
117     return this;
118   }

120   /**
121    * Threadify the writing of the inputStream to disk.
122    *
123    * <p>Internal method and shouldn't be used externally.
124    *
125    * @param inputStream {@link InputStream} from {@link Process#getInputStream()} and {@link
126    *     Process#getErrorStream()}.
127    * @param outputSourceType Type of Output as per {@link OutputSourceType}
128    */
129   private void saveInpuStreamToDisk(InputStream inputStream, OutputSourceType outputSourceType) {
130     inputStreamReadingThreads.add(
131         this.executorService.submit(() -> this.readInputStream(inputStream, outputSourceType)));
132   }

134   /**
135    * Blocking method to wait for all all 3 threads to be finished. 2 threads are to read {@link
136    * OutputSourceType#SYSOUT} and {@link OutputSourceType#SYSERR}. The last thread is to write the
137    * read {@link InputStream} to disk.
138    *
139    * <p>Call this method to wait for the log handler threads to be finished execution.
140    *
141    * @throws InterruptedException when the {@link ExecutorService#awaitTermination(long, TimeUnit)}
142    *     is interrupted by some other {@link Thread}.
143    * @throws ExecutionException when waiting for the disk writer to finish.
144    */
145   public void waitForShutdown() throws InterruptedException, ExecutionException {
146     if (this.configuration.getMasterLogFile() != null) {
147       logger.info("Waiting for all the logs writing thread to shutdown.");
148       //
149       // Wait for the disk writing thread to stop.
150       //
151       this.diskWritingThread.get();
152       //
153       // Wait for the termination of executor service.
154       //
155       this.executorService.awaitTermination(Long.MAX_VALUE, TimeUnit.MINUTES);
156       logger.info("Waiting for all the logs writing thread to shutdown.");
157 
158     } else {
159       logger.error("Masterfile Configuration is missing : {}", this.processConfigrationAsString);
160       logger.error("Discarding logs.");
161     }
162   }

164   /**
165    * Write the content of {@link LogHandler#queue} to the file as recived from {@link Configuration}
166    * object.
167    *
168    * <p>Internal method, shouldn't be used externallyy. Automatically called when a object of {@link
169    * LogHandler} is created.
170    *
171    * @return int depciting the number of lines written.
172    * @throws InterruptedException if the disk writing thread is interrupted.
173    */
174   private int writeToDisk() throws InterruptedException {
175     String threadName =
176         new StringJoiner(EMPTY_STR)
177             .add(Thread.currentThread().getName())
178             .add(ProcessRunnerConstants.DISK_WRITER_THREAD_NAME_SUFFIX)
179             .toString();
180     Thread.currentThread().setName(threadName);
181     logger.info("Starting {}  to write to disk", threadName);
182     //
183     // Look for log while the intputstream still have data or the queue have not
184     // finished writing to
185     // disk
186     //
187     int counter = -1;
188     while (this.inputStreamReadingThreads.stream().anyMatch(future -> !future.isDone())
189         || !queue.isEmpty()) {
190       if (queue.isEmpty()) {
191         //
192         // If queue is empty wait for some time
193         //
194         logger.debug(
195             "Queue is empty, waiting for {} milliseconds", ProcessRunnerConstants.THREAD_WAIT_TIME);
196         Thread.sleep(ProcessRunnerConstants.THREAD_WAIT_TIME);
197       } else {
198         //
199         // Write all the element in the queue to the disk.
200         //
201         List<OutputRecord> record = new ArrayList<>(ProcessRunnerConstants.FILE_WRITER_OBJECT_SIZE);
202         int numberElementDrained =
203             queue.drainTo(record, ProcessRunnerConstants.FILE_WRITER_OBJECT_SIZE);
204         assert numberElementDrained == record.size();
205         counter += record.size();
206         record.stream().map(ProcessRunnerConstants.GSON::toJson).forEach(printWriter::println);
207       }
208       //
209       // Force flush
210       //
211       printWriter.flush();
212     }
213     printWriter.close();
214     logger.debug("Wrote {} lines to master log file.", ++counter);
215     return counter;
216   }

218   /**
219    * Reads the {@link InputStream} and write them to a {@link LogHandler#queue} as {@link
220    * OutputRecord}.
221    *
222    * <p>This is a internal method and shouldn't be used by any body in the library.
223    *
224    * @param inputStream the {@link InputStream} to be read. The inputStream is received from {@link
225    *     Process#getErrorStream()} and {@link Process#getInputStream()}.
226    * @param outputSourceType either as input {@link OutputSourceType#SYSOUT} or {@link
227    *     OutputSourceType#SYSERR}.
228    */
229   private void readInputStream(InputStream inputStream, OutputSourceType outputSourceType) {
230     String outputSourceTypeAsString = outputSourceType.toString();
231     logger.trace("Saving input stream for : {}", outputSourceTypeAsString);
232     String threadName =
233         new StringJoiner("")
234             .add(Thread.currentThread().getName())
235             .add(ProcessRunnerConstants.STREAM_READER_THREAD_NAME_SUFFIX)
236             .add(outputSourceType.toString())
237             .toString();
238     Thread.currentThread().setName(threadName);
239     logger.trace("Starting {} to read {}", threadName, outputSourceTypeAsString);
240     Scanner scanner = new Scanner(inputStream, Charset.defaultCharset().toString());
241     String currentLine;
242     String loggingMessage;
243     while (scanner.hasNextLine()) {
244       currentLine = scanner.nextLine();
245       loggingMessage =
246           new StringJoiner(" >> ").add(outputSourceType.toString()).add(currentLine).toString();
247       if (streamingEnabled) {
248         logger.info(loggingMessage);
249       } else {
250         logger.trace(loggingMessage);
251       }
252       boolean response = this.queue.add(new OutputRecord(outputSourceType, currentLine));
253       assert response;
254     }
255   }
