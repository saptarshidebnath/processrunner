18  public class LogWriter {
19    private static final Logger logger = LoggerFactory.getLogger(GrepFile.class);
20    /**
21     * Write a log content from the {@link Configuration#masterLogFile} to new {@link File} as per
22     * provided {@link Configuration} for a particular {@link OutputSourceType}.
23     *
24     * @param configuration accepts a {@link Configuration}
25     * @param targetFile accepts a target {@link File}.
26     * @param outputSourceType Accepts the {@link OutputSourceType} which need to be printed only.
27     * @return a {@link File} reference to the newly written log {@link File}.
28     * @throws IOException when there are problems reading and wrting the {@link File}.
29     */
30    public File writeLog(
31        final Configuration configuration,
32        final File targetFile,
33        final OutputSourceType outputSourceType)
34        throws IOException {
35      try (FileOutputStream fileOutputStream = new FileOutputStream(targetFile, Boolean.TRUE);
36          PrintWriter printWriter =
37              new PrintWriter(new OutputStreamWriter(fileOutputStream, configuration.getCharset()))) {
38        logger.trace("Writing {} to : {}", outputSourceType, targetFile.getCanonicalPath());
39  
40        try (BufferedReader br =
41            new BufferedReader(
42                new InputStreamReader(
43                    new FileInputStream(configuration.getMasterLogFile()),
44                    configuration.getCharset()))) {
45          for (OutputRecord record;
46              (record = ProcessRunnerConstants.GSON.fromJson(br.readLine(), OutputRecord.class))
47                  != null; ) {
48            if (outputSourceType == OutputSourceType.ALL
49                || outputSourceType == record.getOutputSourceType()) {
50              printWriter.println(record.getOutputText());
51            }
52          }
53        }
54      }
55      logger.info("{} written completely to : {}", outputSourceType, targetFile.getCanonicalPath());
56      return targetFile;
57    }
58  }
