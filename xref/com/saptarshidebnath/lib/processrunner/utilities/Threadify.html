3   import com.saptarshidebnath.lib.processrunner.constants.ProcessRunnerConstants;
4   import java.util.concurrent.ExecutorService;
5   import java.util.concurrent.Executors;
6   import java.util.concurrent.ThreadFactory;
7   import java.util.concurrent.atomic.AtomicInteger;
8   
9   public class Threadify {
10    private static AtomicInteger threadCounter = new AtomicInteger(1);
11  
12    public ExecutorService getProcessRunnerExecutorService() {
13      ThreadFactory processRunnerThreadFactory;
14      processRunnerThreadFactory =
15          runnable ->
16              new Thread(
17                  runnable,
18                  ProcessRunnerConstants.PROCESS_RUNNER_THREAD_GROUP_NAME
19                      + "-"
20                      + threadCounter.getAndIncrement());
21      return Executors.newCachedThreadPool(processRunnerThreadFactory);
22    }
23  }
