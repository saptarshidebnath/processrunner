120   public void seachMasterLogFileForRegex()
121       throws IOException, ExecutionException, ProcessConfigurationException, InterruptedException {
122     final File tempFile = File.createTempFile(FILE_PREFIX_NAME_LOG_DUMP, FILE_SUFFIX_JSON);
123     Output output =
124         RunnerFactory.startProcess(
125             new ConfigBuilder("/bin/bash -c", "ls -la")
126                 .enableLogStreaming(true)
127                 .setMasterLogFile(tempFile, true)
128                 .build());
129     assertThat(
130         "Validating for output", output.grepForRegex(".*").get(0).getOutputText(), is("CNAME"));
131     assertThat(
132         "Validating for output", output.grepForRegex("s[r|o]c").get(0).getOutputText(), is("src"));
133     assertThat(
134         "Validating for output",
135         output.grepForRegex("pom.*").get(0).getOutputText(),

159   @Test
160   public void toStringTest()
161       throws IOException, ExecutionException, ProcessConfigurationException, InterruptedException {
162     final File tempFile = File.createTempFile(FILE_PREFIX_NAME_LOG_DUMP, FILE_SUFFIX_JSON);
163     String interPreter = "/bin/echo";
164     String command = "Saptarshi";
165     String param = "Debnath";
166     Path workingDir = new File(ProcessRunnerConstants.USER_DIR).toPath();
167     Configuration configuration =
168         new ConfigBuilder(interPreter, command)
169             .enableLogStreaming(Boolean.TRUE)
170             .setMasterLogFile(tempFile, Boolean.TRUE)
171             .setParam(param)
172             .setWorkigDir(workingDir)
173             .build();
174     Output output = RunnerFactory.startProcess(configuration);
175     String outputtoString = output.toString();
176     String expectedValue =
177         "OutputImpl{configuration=Configuration{interpreter='"
178             + interPreter
179             + "', command='"
180             + command
181             + " "
182             + param
183             + "', workingDir="
184             + workingDir.toString()
185             + ", masterLogFile="
186             + tempFile.getCanonicalPath()
187             + ", charset=UTF-8, autoDeleteFileOnExit=true, enableLogStreaming=true}, returnCode=0}";
188     assertThat("Output to string", outputtoString, is(expectedValue));
189   }
