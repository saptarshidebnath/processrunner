package com.saptarshidebnath.lib.processrunner.constants;

public enum OutputSourceType {
  /**
   * Depicts the same source similar to {@link System#out}. Please read
   * https://en.wikipedia.org/wiki/Standard_streams for more details.
   */
  SYSOUT,

  /**
   * Depict the same source similar to {@link System#err}. Please read
   * https://en.wikipedia.org/wiki/Standard_streams for more details.
   */
  SYSERR,

  /**
   * Depcts both sysout and SYSERR Please read https://en.wikipedia.org/wiki/Standard_streams for
   * more details.
   */
  ALL
}
